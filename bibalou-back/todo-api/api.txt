List of all the api

To connect a user :
  /authenticate/auth
  req : POST
    login : 'aa.bb@cc.dd'
    password : 'pwd'
  res :
    Success :
      success : true,
      token : 'String'
    Fail :
      {
        success : false,
        message : "User does not exist or wrong password",
        errcode : 2
      }

To register new User with basic rigth
  /authenticate/newUser
  req : POST
    login : 'aa.bb@cc.dd'
    password : 'pwd'
  res :
    Success :
      {
        success : true;
        token : 'String'
      }
    Fail :
    {
      success : false,
      message : 'Email already use',
      errcode : 1
    }
